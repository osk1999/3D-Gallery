#version 400 core

uniform int lightNum;

layout(location = 0) in vec3 position;
in vec2 textureCoords;
in vec3 normalData;

out vec2 pass_tCoords;
out vec3 normVector;
out vec3 toLightVectors[14];
out vec3 toCameraVector;
out vec3 vertPos;

uniform mat4 transformMatrix;
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;

uniform vec3 lightPosition[14];

void main(void)
{
	vec4 worldPosition = transformMatrix * vec4(position, 1.0);
    gl_Position = projectionMatrix * viewMatrix * worldPosition;
    
    pass_tCoords = textureCoords;
    mat3 normalMatrix = transpose(inverse(mat3(transformMatrix)));
	normVector = normalize(normalMatrix * normalData);
    for(int i=0; i < lightNum; i++){
    	vec3 toLightVector = lightPosition[i] - worldPosition.xyz;
    	toLightVectors[i] = toLightVector;
    }
    
   	vertPos = worldPosition.xyz;
    toCameraVector = (inverse(viewMatrix) * vec4(0.0, 0.0, 0.0, 1.0)).xyz - worldPosition.xyz;
}
